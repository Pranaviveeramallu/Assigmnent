Below is a simplified code snippet that demonstrates a basic approach for tracking the lifecycle of products and implementing circular economy principles using a simple example. You will need to adapt and expand this code to fit your specific requirements and blockchain implementation.


import hashlib
import json
from datetime import datetime

# Define the product information
products = [
    {'name': 'Product1', 'id': '1234567890', 'status': 'production'},
    {'name': 'Product2', 'id': '0987654321', 'status': 'distribution'},
    {'name': 'Product3', 'id': '5678901234', 'status': 'consumption'},
    # Add more products as needed
]

# Define the blockchain and initial block
blockchain = []
genesis_block = {
    'index': 0,
    'timestamp': str(datetime.now()),
    'data': 'Genesis Block',
    'previous_hash': '',
    'hash': ''
}
blockchain.append(genesis_block)

def calculate_hash(block):
    """
    Calculate the hash value of a block using SHA-256.
    """
    block_string = json.dumps(block, sort_keys=True).encode()
    return hashlib.sha256(block_string).hexdigest()

def create_block(data, previous_hash):
    """
    Create a new block for the blockchain.
    """
    block = {
        'index': len(blockchain),
        'timestamp': str(datetime.now()),
        'data': data,
        'previous_hash': previous_hash,
        'hash': ''
    }
    block['hash'] = calculate_hash(block)
    return block

def add_block_to_chain(block):
    """
    Add a new block to the blockchain.
    """
    blockchain.append(block)

def circular_economy_integration(product):
    """
    Perform circular economy integration for a specific product.
    """
    # Get the latest block from the blockchain
    previous_block = blockchain[-1]

    # Create data for the new block representing the product's lifecycle update
    data = {
        'product_id': product['id'],
        'timestamp': str(datetime.now()),
        'status': product['status']
    }

    # Create a new block for the updated product lifecycle and add it to the blockchain
    new_block = create_block(data, previous_block['hash'])
    add_block_to_chain(new_block)

# Example usage:
product = products[0]
product['status'] = 'distribution'
circular_economy_integration(product)

# Print the updated blockchain
for block in blockchain:
    print(block)
