import random

# Define the supplier information
suppliers = [
    {'name': 'Supplier1', 'id': '123456', 'ethical_standards': 'High'},
    {'name': 'Supplier2', 'id': '789012', 'ethical_standards': 'Medium'},
    {'name': 'Supplier3', 'id': '345678', 'ethical_standards': 'Low'},
    # Add more suppliers as needed
]

# Define the supplier rating information
supplier_ratings = {}

def assess_supplier(supplier_id, ethical_standards):
    """
    Assess and rate a supplier based on their ethical standards.
    """
    if supplier_id not in supplier_ratings:
        supplier_ratings[supplier_id] = {}
    
    supplier_ratings[supplier_id]['ethical_standards'] = ethical_standards

def get_supplier_rating(supplier_id):
    """
    Get the rating information for a specific supplier.
    """
    if supplier_id in supplier_ratings:
        return supplier_ratings[supplier_id]
    else:
        return {}

# Example usage:
supplier = random.choice(suppliers)
supplier_id = supplier['id']
ethical_standards = supplier['ethical_standards']
assess_supplier(supplier_id, ethical_standards)

# Get the rating information for the supplier
supplier_rating = get_supplier_rating(supplier_id)
print(f"Rating for Supplier {supplier_id}:")
if supplier_rating:
    ethical_standards_rating = supplier_rating.get('ethical_standards', 'Not rated')
    print(f"Ethical Standards: {ethical_standards_rating}")
else:
    print("No rating available for the supplier")
