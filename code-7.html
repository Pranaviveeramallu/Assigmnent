import random

# Define the product information
products = [
    {'name': 'Product1', 'id': '123456', 'packaging_material': 'Cardboard'},
    {'name': 'Product2', 'id': '789012', 'packaging_material': 'Biodegradable Plastic'},
    {'name': 'Product3', 'id': '345678', 'packaging_material': 'Recycled Paper'},
    # Add more products as needed
]

# Define the packaging tracking information
packaging_info = {}

def track_packaging(product_id, packaging_material):
    """
    Track the packaging information for a specific product.
    """
    if product_id not in packaging_info:
        packaging_info[product_id] = []
    
    packaging_info[product_id].append(packaging_material)

def get_packaging_history(product_id):
    """
    Get the packaging history for a specific product.
    """
    if product_id in packaging_info:
        return packaging_info[product_id]
    else:
        return []

# Example usage:
product = random.choice(products)
product_id = product['id']
packaging_material = product['packaging_material']
track_packaging(product_id, packaging_material)

# Get the packaging history for the product
packaging_history = get_packaging_history(product_id)
print(f"Packaging History for Product {product_id}:")
for i, material in enumerate(packaging_history):
    print(f"Packaging {i+1}: {material}")
